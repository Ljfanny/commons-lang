30 
 * <pre>

32 
 *   String title;
33 
 *   ...
34 
 * }
35 
 *
36 
 * public class Person {
37 
 *   String name;
38 
 *   int age;
39 
 *   boolean smoker;
40 
 *   Job job;
41 
 *
42 
 *   ...

 *
50 
 * <p>
51 
 * This will produce a toString of the format:<br>
52

55 
 * &nbsp; smokealse, <br>
56 
 * &nbsp; job=Job@43cd2[ <br>
57 
 * &nbsp; &nbsp; title=Manager <br>
58 
 * &nbsp;  ] <br>

 * ]

 * </code>
